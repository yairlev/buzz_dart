Code void set length(int newLength) #

void set length(int newLength) {
  final len = this.length;
  if (newLength >= len) {
    return;
  } else if (newLength < 0) {
    throw const IllegalArgumentException("Invalid list length");
  }

  removeRange(newLength - 1, len - newLength);
}
